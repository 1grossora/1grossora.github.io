from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestRegressor
from sklearn.model_selection import RandomizedSearchCV
import pandas as pd
import matplotlib.pylab as plt
# This makes our model... the range is chose from prior guesses
def make_model(myX, myy):
    myregr = RandomForestRegressor()
    param_grid = { 
    'n_estimators': [500,800,1000],
    'max_features': ['auto','sqrt'],
    'max_depth': [6,8,10]
     }
    CV_rfc = RandomizedSearchCV(estimator=myregr, param_distributions=param_grid,cv=5)
    #print len(bX_train.columns)
    CV_rfc.fit(myX, myy)
    
    pv = CV_rfc.best_params_
    print pv

    regr = RandomForestRegressor(random_state=0,n_jobs=-1,n_estimators=pv.get('n_estimators'),max_features=pv.get('max_features'),max_depth=pv.get('max_depth'))
    regr.fit(myX,myy)
    return regr

def make_model_resolution(mymodel, myx, myy, **kwargs ):
    testdiff = np.subtract(np.asarray(myy),np.asarray(mymodel.predict(myx)))
    plt.hist(testdiff,bins=200,range=(-0.03,0.03), **kwargs)
    plt.show()
def model_stats(mymodel, myx, myy, **kwargs ):
    testdiff = np.subtract(np.asarray(myy),np.asarray(mymodel.predict(myx)))
    return testdiff.mean(),testdiff.std()

def model_sensitivity_nlikes(mymodel, myx, myy, N_followers ,mystd=1.96 ):
    testdiff = np.subtract(np.asarray(myy),np.asarray(mymodel.predict(myx)))
    spread  = testdiff.std()* mystd *N_followers*2
    return spread
